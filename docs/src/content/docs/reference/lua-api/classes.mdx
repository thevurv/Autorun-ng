import { Aside, Card, Code, Tabs, TabItem } from '@astrojs/starlight/components';
import RealmBadge from '../../../../components/RealmBadge.astro';
import apiData from '../../../../../../packages/autorun-env/api.json';
import { formatParameters, formatReturns } from '../../../../utils/api-formatter';

{apiData.classes && apiData.classes.map(cls => (
  <div key={cls.name}>
    <h2><RealmBadge realm={cls.realm} /> <code>{cls.name}</code></h2>
    <p>{cls.description}</p>

    {cls.fields && cls.fields.length > 0 && (
      <>
        <h3>Fields</h3>
        {cls.fields.map((field, index) => (
          <div key={field.name}>
            <h4><RealmBadge realm={field.realm} /> <code>{field.name}: {field.type}</code></h4>
            <p>{field.description}</p>
            {index < cls.fields.length - 1 && <hr style={{ margin: '2rem 0' }} />}
          </div>
        ))}
      </>
    )}

    {cls.methods && cls.methods.length > 0 && (
      <>
        <h3>Methods</h3>
        <hr />
        <div style={{ display: "flex", flexDirection: "column", gap: "2.5rem" }}>
        {cls.methods.map(method => (
          <div key={method.name}>
            <h4><RealmBadge realm={method.realm} /> <code>{method.name}({formatParameters(method.parameters)}): {formatReturns(method.returns)}</code></h4>
            <p>{method.description}</p>

            {method.parameters && method.parameters.length > 0 && (
              <>
                <p><strong>Parameters:</strong></p>
                <ul>
                  {method.parameters.map((param, i) => (
                    <li key={i}>
                      <code>{param.name}</code> (<code>{param.type}</code>)
                      {param.description && ` - ${param.description}`}
                    </li>
                  ))}
                </ul>
              </>
            )}

            {method.returns && method.returns.length > 0 && (
              <>
                <p><strong>Returns:</strong></p>
                <ul>
                  {method.returns.map((ret, i) => (
                    <li key={i}>
                      <code>{ret.type}</code>
                      {ret.description && ` - ${ret.description}`}
                    </li>
                  ))}
                </ul>
              </>
            )}
          </div>
        ))}
        </div>
      </>
    )}
  </div>
))}
